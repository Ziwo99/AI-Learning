Création d'un jeu en console

Dans cette sous-section, vous apprendrez à créer un jeu en console en utilisant le langage C. Vous explorerez les concepts de base de la programmation de jeux, tels que la gestion des entrées utilisateur, la logique du jeu et l'affichage des résultats.

Voici un exemple de code pour créer un jeu simple en console :

#include <stdio.h>

int main() {
    int nombreMystere = 42;
    int nombreUtilisateur;

    printf("Devinez le nombre mystere !\n");

    do {
        printf("Entrez un nombre : ");
        scanf("%d", &nombreUtilisateur);

        if (nombreUtilisateur < nombreMystere) {
            printf("C'est plus !\n");
        } else if (nombreUtilisateur > nombreMystere) {
            printf("C'est moins !\n");
        } else {
            printf("Bravo, vous avez trouve le nombre mystere !\n");
        }
    } while (nombreUtilisateur != nombreMystere);

    return 0;
}

Développement d'une application de gestion de données

Dans cette sous-section, vous apprendrez à développer une application de gestion de données en utilisant le langage C. Vous découvrirez comment créer, lire, mettre à jour et supprimer des données à partir d'un fichier.

Voici un exemple de code pour créer une application de gestion de contacts :

#include <stdio.h>
#include <stdlib.h>
#include <string.h>

typedef struct {
    char nom[50];
    char email[50];
    char telephone[15];
} Contact;

int main() {
    FILE *fichier;
    Contact contact;
    char choix;

    fichier = fopen("contacts.txt", "a+");

    if (fichier == NULL) {
        printf("Erreur lors de l'ouverture du fichier.\n");
        return 1;
    }

    do {
        printf("Entrez le nom du contact : ");
        scanf("%s", contact.nom);

        printf("Entrez l'email du contact : ");
        scanf("%s", contact.email);

        printf("Entrez le telephone du contact : ");
        scanf("%s", contact.telephone);

        fwrite(&contact, sizeof(Contact), 1, fichier);

        printf("Voulez-vous ajouter un autre contact ? (o/n) ");
        scanf(" %c", &choix);
    } while (choix == 'o' || choix == 'O');

    fclose(fichier);

    return 0;
}

Implémentation d'un algorithme complexe

Dans cette sous-section, vous apprendrez à implémenter un algorithme complexe en utilisant le langage C. Vous explorerez des algorithmes avancés tels que le tri, la recherche et les algorithmes de graphes.

Voici un exemple de code pour implémenter l'algorithme de tri rapide (quicksort) :

#include <stdio.h>

void swap(int *a, int *b) {
    int temp = *a;
    *a = *b;
    *b = temp;
}

int partition(int arr[], int low, int high) {
    int pivot = arr[high];
    int i = (low - 1);

    for (int j = low; j <= high - 1; j++) {
        if (arr[j] < pivot) {
            i++;
            swap(&arr[i], &arr[j]);
        }
    }

    swap(&arr[i + 1], &arr[high]);
    return (i + 1);
}

void quicksort(int arr[], int low, int high) {
    if (low < high) {
        int pi = partition(arr, low, high);

        quicksort(arr, low, pi - 1);
        quicksort(arr, pi + 1, high);
    }
}

int main() {
    int arr[] = {64, 34, 25, 12, 22, 11, 90};
    int n = sizeof(arr) / sizeof(arr[0]);

    quicksort(arr, 0, n - 1);

    printf("Le tableau trie : \n");
    for (int i = 0; i < n; i++) {
        printf("%d ", arr[i]);
    }

    return 0;
}

Assurez-vous de comprendre chaque ligne de code et de l'adapter à vos besoins spécifiques lorsque vous créez vos propres projets pratiques en C.
