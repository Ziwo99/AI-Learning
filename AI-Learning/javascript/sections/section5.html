Section 5: Les API Web en Javascript

Les API de géolocalisation

L'API de géolocalisation permet aux applications web d'accéder à la position géographique de l'utilisateur. Cela peut être utilisé pour fournir des fonctionnalités basées sur la localisation, telles que des services de localisation, des recommandations personnalisées et bien plus encore.

Voici un exemple de code qui utilise l'API de géolocalisation pour obtenir la position de l'utilisateur :

if (navigator.geolocation) {
    navigator.geolocation.getCurrentPosition(function(position) {
        var latitude = position.coords.latitude;
        var longitude = position.coords.longitude;
        console.log("Latitude: " + latitude + ", Longitude: " + longitude);
    });
} else {
    console.log("La géolocalisation n'est pas prise en charge par ce navigateur.");
}

Les API de caméra et de microphone

Les API de caméra et de microphone permettent aux applications web d'accéder aux fonctionnalités de la caméra et du microphone de l'appareil de l'utilisateur. Cela peut être utilisé pour créer des applications de chat vidéo, des enregistreurs audio, des applications de réalité augmentée, etc.

Voici un exemple de code qui utilise l'API de caméra pour prendre une photo :

var video = document.querySelector('video');

navigator.mediaDevices.getUserMedia({ video: true })
    .then(function(stream) {
        video.srcObject = stream;
        video.play();
    })
    .catch(function(error) {
        console.log("Une erreur s'est produite lors de l'accès à la caméra: " + error);
    });

var canvas = document.querySelector('canvas');
var context = canvas.getContext('2d');

document.querySelector('button').addEventListener('click', function() {
    context.drawImage(video, 0, 0, canvas.width, canvas.height);
});

Les API de stockage local

Les API de stockage local permettent aux applications web de stocker des données localement sur l'appareil de l'utilisateur. Cela peut être utilisé pour créer des applications hors ligne, des applications de gestion de données, etc.

Voici un exemple de code qui utilise l'API de stockage local pour enregistrer des données :

localStorage.setItem('nom', 'John');
localStorage.setItem('age', 30);

var nom = localStorage.getItem('nom');
var age = localStorage.getItem('age');

console.log("Nom: " + nom + ", Age: " + age);

Les API de notifications

Les API de notifications permettent aux applications web d'afficher des notifications à l'utilisateur, même lorsque l'application n'est pas ouverte. Cela peut être utilisé pour envoyer des notifications push, des rappels, des mises à jour, etc.

Voici un exemple de code qui utilise l'API de notifications pour afficher une notification :

if (Notification.permission === 'granted') {
    var notification = new Notification('Nouvelle notification', {
        body: 'Ceci est une notification de test.',
        icon: 'icon.png'
    });
} else if (Notification.permission !== 'denied') {
    Notification.requestPermission(function(permission) {
        if (permission === 'granted') {
            var notification = new Notification('Nouvelle notification', {
                body: 'Ceci est une notification de test.',
                icon: 'icon.png'
            });
        }
    });
}
