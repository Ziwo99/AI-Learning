Section 12 : Projets pratiques en Javascript

Projet 1 : Création d'un jeu en Javascript

Dans ce projet, nous allons créer un jeu simple en utilisant Javascript. Nous allons apprendre à manipuler les éléments HTML, à gérer les événements et à mettre en place la logique du jeu.

Voici un exemple de code pour créer un jeu de devinette en Javascript :

let randomNumber = Math.floor(Math.random() * 100) + 1;
let guesses = 0;

document.querySelector('#check-button').addEventListener('click', function() {
    let userGuess = Number(document.querySelector('#guess-input').value);
    guesses++;

    if (userGuess === randomNumber) {
        document.querySelector('#result').textContent = 'Bravo, vous avez deviné le nombre en ' + guesses + ' essais !';
    } else if (userGuess > randomNumber) {
        document.querySelector('#result').textContent = 'Trop grand, essayez encore.';
    } else {
        document.querySelector('#result').textContent = 'Trop petit, essayez encore.';
    }
});

Projet 2 : Développement d'une application web en Javascript

Dans ce projet, nous allons apprendre à développer une application web en utilisant Javascript. Nous allons explorer les concepts de manipulation du DOM, de gestion des événements et de communication avec un serveur.

Voici un exemple de code pour créer une application web de liste de tâches en Javascript :

let taskList = [];

document.querySelector('#add-button').addEventListener('click', function() {
    let taskInput = document.querySelector('#task-input').value;
    taskList.push(taskInput);
    displayTasks();
});

function displayTasks() {
    let taskListElement = document.querySelector('#task-list');
    taskListElement.innerHTML = '';

    for (let i = 0; i < taskList.length; i++) {
        let taskItem = document.createElement('li');
        taskItem.textContent = taskList[i];
        taskListElement.appendChild(taskItem);
    }
}

Projet 4 : Réalisation d'un site e-commerce en Javascript

Dans ce projet, nous allons apprendre à créer un site e-commerce en utilisant Javascript. Nous allons explorer les concepts de manipulation du DOM, de gestion des données et de mise en place d'un panier d'achat.

Voici un exemple de code pour créer un site e-commerce en Javascript :

let products = [
    { id: 1, name: 'Product 1', price: 10 },
    { id: 2, name: 'Product 2', price: 20 },
    { id: 3, name: 'Product 3', price: 30 }
];

let cart = [];

function addToCart(productId) {
    let product = products.find(item => item.id === productId);
    cart.push(product);
}

function displayCart() {
    let cartElement = document.querySelector('#cart');
    cartElement.innerHTML = '';

    for (let i = 0; i < cart.length; i++) {
        let cartItem = document.createElement('li');
        cartItem.textContent = cart[i].name + ' - $' + cart[i].price;
        cartElement.appendChild(cartItem);
    }
}

Projet 5 : Mise en place d'un chatbot en Javascript

Dans ce projet, nous allons apprendre à créer un chatbot en utilisant Javascript. Nous allons explorer les concepts de traitement du langage naturel, de gestion des conversations et de mise en place d'une interface utilisateur pour le chatbot.

Voici un exemple de code pour créer un chatbot en Javascript :

const chatbot = {
    greeting: 'Bonjour, comment puis-je vous aider ?',
    responses: {
        'Bonjour': 'Bonjour, comment puis-je vous aider ?',
        'Quel est votre nom ?': 'Je suis un chatbot.',
        'Quel temps fait-il aujourd\'hui ?': 'Il fait beau et ensoleillé.',
        'Au revoir': 'Au revoir, à bientôt !'
    },
    getResponse: function(message) {
        return this.responses[message] || 'Désolé, je ne comprends pas.';
    }
};

function sendMessage() {
    let messageInput = document.querySelector('#message-input');
    let message = messageInput.value;
    let chatbox = document.querySelector('#chatbox');
    let response = chatbot.getResponse(message);

    let messageElement = document.createElement('div');
    messageElement.textContent = 'User: ' + message;
    chatbox.appendChild(messageElement);

    let responseElement = document.createElement('div');
    responseElement.textContent = 'Chatbot: ' + response;
    chatbox.appendChild(responseElement);

    messageInput.value = '';
}
